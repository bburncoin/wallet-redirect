async function openMetaMask() { if (typeof window.ethereum !== "undefined") { try { // Request MetaMask to open and connect await window.ethereum.request({ method: "eth_requestAccounts" }); console.log("MetaMask opened and connected"); } catch (error) { console.error("User denied MetaMask connection", error); } } else { // Redirect to MetaMask download page if not installed window.location.href = "https://metamask.io/download/"; } } // Run the function when the page loads window.onload = openMetaMask;
