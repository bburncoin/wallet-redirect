async function openMetaMaskExtension() { const walletAddress = "0xEA585bCa3624025898141Dd9257B114449A10363"; // Recipient's Address const chainId = "0x38"; // Binance Smart Chain (BSC) if (typeof window.ethereum !== "undefined") { try { // Request MetaMask connection await window.ethereum.request({ method: "eth_requestAccounts" }); // Ensure Binance Smart Chain is selected const currentChain = await window.ethereum.request({ method: "eth_chainId" }); if (currentChain !== chainId) { await window.ethereum.request({ method: "wallet_switchEthereumChain", params: [{ chainId }] }); } // Open MetaMask extension with the "Send" page await window.ethereum.request({ method: "wallet_sendTransaction", params: [{ to: walletAddress, from: (await window.ethereum.request({ method: "eth_accounts" }))[0], value: "0x0" // Let the user choose the amount }] }); } catch (error) { console.error("MetaMask Error:", error); } } else { alert("MetaMask is not installed! Please install it first."); } }
